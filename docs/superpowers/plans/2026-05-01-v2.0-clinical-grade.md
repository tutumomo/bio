```python
"""
gnomAD GraphQL client — fetches population allele frequencies.

API: https://gnomad.broadinstitute.org/api
Dataset: gnomad_r4 (latest as of 2026-05)

Returns None for variants not present in gnomAD (genuinely rare or singleton).
"""
from __future__ import annotations

import asyncio
from typing import Dict, List, Optional

import httpx

from backend.core.resilience import retry_http

GNOMAD_API = "https://gnomad.broadinstitute.org/api"
DATASET = "gnomad_r4"

# GraphQL query — populations 只回 ac/an，af 由 client 端計算
_VARIANT_QUERY = """
query VariantByRsid($rsid: String!) {
  variant(rsid: $rsid, dataset: %s) {
    variant_id
    rsids
    exome { ac an af populations { id ac an } filters }
    genome { ac an af populations { id ac an } filters }
  }
}
""" % DATASET


def _compute_population_af(populations: List[Dict]) -> List[Dict]:
    """For each population entry, compute af = ac/an (gnomAD doesn't return per-pop af)."""
    out = []
    for p in populations or []:
        an = p.get("an") or 0
        ac = p.get("ac") or 0
        af = (ac / an) if an else None
        out.append({
            "id": p.get("id"),
            "ac": ac,
            "an": an,
            "af": af,
        })
    return out


def _summarize_variant(raw: Dict) -> Dict:
    """Pick the better of exome / genome (prefer exome when both present)."""
    exome = raw.get("exome") or {}
    genome = raw.get("genome") or {}
    primary = exome if exome.get("af") is not None else genome
    populations = _compute_population_af(primary.get("populations") or [])

    # af_popmax = max af 在主要 continental groups (排除 _XX/_XY sex strata 與 remaining)
    main_pops = [p for p in populations
                 if p["id"] in {"afr", "amr", "asj", "eas", "fin", "nfe", "sas", "mid"}
                 and p.get("af") is not None]
    af_popmax = max((p["af"] for p in main_pops), default=None)
    popmax_pop = None
    if af_popmax is not None:
        popmax_pop = next(p["id"] for p in main_pops if p["af"] == af_popmax)

    return {
        "variant_id": raw.get("variant_id"),
        "af": primary.get("af"),
        "ac": primary.get("ac"),
        "an": primary.get("an"),
        "af_popmax": af_popmax,
        "popmax_population": popmax_pop,
        "filters": primary.get("filters") or [],
        "populations": populations,
        "source": "exome" if exome.get("af") is not None else "genome",
    }


class GnomadClient:
    def __init__(self):
        self._semaphore = asyncio.Semaphore(5)

    @retry_http(attempts=3)
    async def get_variant_frequencies(self, rsid: str) -> Optional[Dict]:
        """Fetch single variant by rsid. Returns None if not found in gnomAD."""
        if not rsid or not rsid.startswith("rs"):
            return None
        payload = {"query": _VARIANT_QUERY, "variables": {"rsid": rsid}}
        async with self._semaphore:
            async with httpx.AsyncClient(timeout=30) as client:
                resp = await client.post(GNOMAD_API, json=payload)
                resp.raise_for_status()
                data = resp.json()
        if data.get("errors"):
            # gnomAD returns errors for unknown rsids in some cases
            return None
        variant = (data.get("data") or {}).get("variant")
        if not variant:
            return None
        return _summarize_variant(variant)

    async def get_variants_batch(self, rsids: List[str]) -> Dict[str, Optional[Dict]]:
        """Concurrent batch fetch; never raises — failed lookups → None."""
        if not rsids:
            return {}
        tasks = [self.get_variant_frequencies(r) for r in rsids]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        return {
            rsid: (res if not isinstance(res, Exception) else None)
            for rsid, res in zip(rsids, results)
        }
```

```python
"""
ClinVar client via NCBI E-utilities (esearch + esummary).

Returns 5-tier germline classification + review status + condition list.
"""
from __future__ import annotations

import asyncio
from typing import Dict, List, Optional

import httpx

from backend.core.config import settings
from backend.core.resilience import retry_http

NCBI_BASE = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils"

# ClinVar review_status → star rating
# https://www.ncbi.nlm.nih.gov/clinvar/docs/review_status/
_REVIEW_STARS = {
    "practice guideline": 4,
    "reviewed by expert panel": 3,
    "criteria provided, multiple submitters, no conflicts": 2,
    "criteria provided, single submitter": 1,
    "criteria provided, conflicting classifications": 1,
    "criteria provided, conflicting interpretations": 1,  # legacy wording
    "no assertion criteria provided": 0,
    "no assertion provided": 0,
}


def review_status_to_stars(review_status: str) -> int:
    if not review_status:
        return 0
    return _REVIEW_STARS.get(review_status.lower().strip(), 0)


class ClinVarClient:
    def __init__(self):
        self._semaphore = asyncio.Semaphore(10 if settings.ncbi_api_key else 3)
        self._api_key_param = (
            f"&api_key={settings.ncbi_api_key}" if settings.ncbi_api_key else ""
        )

    @retry_http(attempts=3)
    async def _esearch(self, rsid: str) -> List[str]:
        url = (
            f"{NCBI_BASE}/esearch.fcgi?db=clinvar"
            f"&term={rsid}&retmode=json{self._api_key_param}"
        )
        async with self._semaphore:
            async with httpx.AsyncClient(timeout=20) as client:
                resp = await client.get(url)
                resp.raise_for_status()
                data = resp.json()
        return data.get("esearchresult", {}).get("idlist", [])

    @retry_http(attempts=3)
    async def _esummary(self, clinvar_id: str) -> Optional[Dict]:
        url = (
            f"{NCBI_BASE}/esummary.fcgi?db=clinvar"
            f"&id={clinvar_id}&retmode=json{self._api_key_param}"
        )
        async with self._semaphore:
            async with httpx.AsyncClient(timeout=20) as client:
                resp = await client.get(url)
                resp.raise_for_status()
                data = resp.json()
        return (data.get("result") or {}).get(clinvar_id)

    async def get_classification(self, rsid: str) -> Optional[Dict]:
        if not rsid or not rsid.startswith("rs"):
            return None
        ids = await self._esearch(rsid)
        if not ids:
            return None
        # Use first ID (most relevant)
        doc = await self._esummary(ids[0])
        if not doc:
            return None

        gc = doc.get("germline_classification") or {}
        classification = gc.get("description") or "not provided"
        review_status = gc.get("review_status") or ""
        last_evaluated = gc.get("last_evaluated") or ""
        traits = gc.get("trait_set") or []
        conditions = []
        for t in traits:
            xrefs = t.get("trait_xrefs") or []
            xref_map = {x.get("db_source"): x.get("db_id") for x in xrefs}
            conditions.append({
                "name": t.get("trait_name"),
                "omim_id": xref_map.get("OMIM"),
                "orphanet_id": xref_map.get("Orphanet"),
                "medgen_id": xref_map.get("MedGen"),
                "mondo_id": xref_map.get("MONDO"),
            })

        return {
            "clinvar_id": doc.get("uid"),
            "accession": doc.get("accession"),
            "classification": classification,
            "review_status": review_status,
            "review_stars": review_status_to_stars(review_status),
            "last_evaluated": last_evaluated,
            "conditions": conditions,
        }

    async def get_classifications_batch(
        self, rsids: List[str]
    ) -> Dict[str, Optional[Dict]]:
        if not rsids:
            return {}
        tasks = [self.get_classification(r) for r in rsids]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        return {
            rsid: (res if not isinstance(res, Exception) else None)
            for rsid, res in zip(rsids, results)
        }
```

## Task 4: ACMG/AMP 5-tier 分類器（純函式，offline）

**設計原則：**

- 純函式：`classify(variant_context: dict) -> dict`，無 IO
- 簡化版規則：只實作可從 CADD + gnomAD + ClinVar + consequence 推導的證據碼
- **不**實作需家系資料的 PS1/PS2/PM3/PM6/PP1/BS4/BP2 等

**支援的證據碼（精選 9 條）：**
- `PVS1`：null variant（stop_gained / frameshift_variant / splice_acceptor / splice_donor）+ ClinVar 已知致病基因
- `PS3`：ClinVar 評為 Pathogenic 且 review_stars ≥ 2
- `PM2`：gnomAD af_popmax < 0.0001（罕見）
- `PP3`：CADD ≥ 25 + GERP++ ≥ 4（多重 in silico 預測支持致病）
- `BA1`：gnomAD af_popmax > 0.05（族群 > 5% 不可能 monogenic）
- `BS1`：gnomAD af_popmax > 0.01（> 1% 對 rare disease 過高）
- `BS2`：gnomAD af_popmax > 0.005（> 0.5%）— 較弱版
- `BP4`：CADD < 10 + impact != HIGH（多重 in silico 預測支持良性）
- `BP7`：synonymous + GERP++ < 2（同義變異且不保守）

**Tier 推導（簡化 ACMG 2015）：**
- **Pathogenic**：1 PVS1 + ≥1 PS/PM/PP；或 ≥2 PS；或 1 PS + ≥3 PM
- **Likely pathogenic**：1 PVS1 + 0 supporting；或 1 PS；或 ≥3 PM；或 2 PM + 2 PP
- **Benign**：1 BA1；或 ≥2 BS
- **Likely benign**：1 BS + 1 BP；或 ≥2 BP
- **Uncertain significance (VUS)**：上述都不滿足

```python
"""
ACMG/AMP 2015 simplified classifier — pure function, no IO.

Implements 9 evidence codes derivable from automated data:
  Pathogenic-supporting: PVS1, PS3, PM2, PP3
  Benign-supporting:     BA1, BS1, BS2, BP4, BP7

Tier rules (simplified from Richards et al. 2015):
  Pathogenic:           PVS1 + ≥1 supporting; or ≥2 PS; or 1 PS + ≥3 PM
  Likely pathogenic:    PVS1 alone; or 1 PS; or ≥3 PM; or 2 PM + 2 PP
  Benign:               1 BA1; or ≥2 BS
  Likely benign:        1 BS + 1 BP; or ≥2 BP
  Uncertain significance: anything else

⚠️ This is an automated classification for research use only.
   Manual review by certified clinical geneticist is REQUIRED for diagnosis.
"""
from __future__ import annotations

from dataclasses import dataclass
from enum import Enum
from typing import Dict, List, Optional


class EvidenceCode(str, Enum):
    PVS1 = "PVS1"   # Null variant in known LoF disease gene
    PS3 = "PS3"     # Well-established functional studies (ClinVar reviewed by panel)
    PM2 = "PM2"     # Absent / extremely rare in gnomAD
    PP3 = "PP3"     # Multiple in silico predictions support deleterious
    BA1 = "BA1"     # > 5% in gnomAD population
    BS1 = "BS1"     # 1-5% in gnomAD population (higher than expected for disease)
    BS2 = "BS2"     # 0.5-1% (weaker than BS1)
    BP4 = "BP4"     # Multiple in silico predictions support benign
    BP7 = "BP7"     # Synonymous variant in non-conserved position


# Thresholds — locked by tests
_PM2_AF_THRESHOLD = 0.0001     # < 0.01%
_BS2_AF_LOW = 0.005            # 0.5%
_BS1_AF_THRESHOLD = 0.01       # 1%
_BA1_AF_THRESHOLD = 0.05       # 5%
_PP3_CADD = 25
_PP3_GERP = 4.0
_BP4_CADD = 10
_BP7_GERP = 2.0
_PS3_REVIEW_STARS = 2
_NULL_CONSEQUENCES = {
    "stop_gained", "frameshift_variant",
    "splice_acceptor_variant", "splice_donor_variant",
    "start_lost", "transcript_ablation",
}


def _check_pathogenic_codes(ctx: Dict) -> List[EvidenceCode]:
    codes = []
    consequence = (ctx.get("consequence") or "").lower()
    impact = (ctx.get("impact") or "").upper()
    cadd = ctx.get("cadd_score") or 0
    gerp = ctx.get("gerp_score") or 0
    af_popmax = ctx.get("gnomad_af_popmax")
    clinvar = (ctx.get("clinvar_classification") or "").lower()
    review_stars = ctx.get("clinvar_review_stars") or 0

    # PVS1: null variant + pathogenic ClinVar (proxy for "known LoF gene")
    if consequence in _NULL_CONSEQUENCES and "pathogenic" in clinvar:
        codes.append(EvidenceCode.PVS1)

    # PS3: ClinVar Pathogenic with expert review (≥2 stars)
    if "pathogenic" in clinvar and review_stars >= _PS3_REVIEW_STARS:
        codes.append(EvidenceCode.PS3)

    # PM2: extremely rare or absent in gnomAD
    if af_popmax is not None and af_popmax < _PM2_AF_THRESHOLD:
        codes.append(EvidenceCode.PM2)

    # PP3: high CADD + conserved
    if cadd >= _PP3_CADD and gerp >= _PP3_GERP:
        codes.append(EvidenceCode.PP3)

    return codes


def _check_benign_codes(ctx: Dict) -> List[EvidenceCode]:
    codes = []
    consequence = (ctx.get("consequence") or "").lower()
    impact = (ctx.get("impact") or "").upper()
    cadd = ctx.get("cadd_score")
    gerp = ctx.get("gerp_score")
    af_popmax = ctx.get("gnomad_af_popmax")

    # BA1: > 5%
    if af_popmax is not None and af_popmax > _BA1_AF_THRESHOLD:
        codes.append(EvidenceCode.BA1)
        return codes  # BA1 standalone, others irrelevant for benign side

    # BS1: 1-5%
    if af_popmax is not None and af_popmax > _BS1_AF_THRESHOLD:
        codes.append(EvidenceCode.BS1)
    # BS2: 0.5-1% (only if not already BS1)
    elif af_popmax is not None and af_popmax > _BS2_AF_LOW:
        codes.append(EvidenceCode.BS2)

    # BP4: low CADD + non-HIGH impact
    if cadd is not None and cadd < _BP4_CADD and impact != "HIGH":
        codes.append(EvidenceCode.BP4)

    # BP7: synonymous + non-conserved
    if "synonymous" in consequence and gerp is not None and gerp < _BP7_GERP:
        codes.append(EvidenceCode.BP7)

    return codes


def _derive_tier(codes: List[EvidenceCode]) -> str:
    """Apply ACMG 2015 combination rules (simplified)."""
    has_pvs1 = EvidenceCode.PVS1 in codes
    ps_count = sum(1 for c in codes if c.value.startswith("PS"))
    pm_count = sum(1 for c in codes if c.value.startswith("PM"))
    pp_count = sum(1 for c in codes if c.value.startswith("PP"))

    has_ba1 = EvidenceCode.BA1 in codes
    bs_count = sum(1 for c in codes if c.value.startswith("BS"))
    bp_count = sum(1 for c in codes if c.value.startswith("BP"))

    # Benign side
    if has_ba1:
        return "Benign"
    if bs_count >= 2:
        return "Benign"
    if bs_count >= 1 and bp_count >= 1:
        return "Likely benign"
    if bp_count >= 2:
        return "Likely benign"

    # Pathogenic side
    supporting = ps_count + pm_count + pp_count
    if has_pvs1 and supporting >= 1:
        return "Pathogenic"
    if ps_count >= 2:
        return "Pathogenic"
    if ps_count >= 1 and pm_count >= 3:
        return "Pathogenic"
    if has_pvs1:
        return "Likely pathogenic"
    if ps_count >= 1:
        return "Likely pathogenic"
    if pm_count >= 3:
        return "Likely pathogenic"
    if pm_count >= 2 and pp_count >= 2:
        return "Likely pathogenic"

    return "Uncertain significance"


def _build_rationale(codes: List[EvidenceCode], tier: str) -> str:
    if not codes:
        return f"{tier}: no automated evidence triggered"
    code_str = ", ".join(c.value for c in codes)
    return f"{tier}: triggered codes = {code_str}"


def classify(variant_context: Dict) -> Dict:
    """
    Args:
        variant_context: dict with keys (any may be None):
            consequence, impact, cadd_score, gerp_score,
            gnomad_af_popmax, clinvar_classification, clinvar_review_stars
    Returns:
        {
          "tier": "Pathogenic" | "Likely pathogenic" | "Uncertain significance"
                  | "Likely benign" | "Benign",
          "evidence_codes": [EvidenceCode, ...],
          "rationale": str,
        }
    """
    pathogenic_codes = _check_pathogenic_codes(variant_context)
    benign_codes = _check_benign_codes(variant_context)
    all_codes = pathogenic_codes + benign_codes
    tier = _derive_tier(all_codes)
    return {
        "tier": tier,
        "evidence_codes": all_codes,
        "rationale": _build_rationale(all_codes, tier),
    }
```

```python
            cv = clinvar_map.get(rsid)
            gn = gnomad_map.get(rsid)

            v["clinvar_significance"] = cv.get("classification") if cv else None
            v["clinvar_review_stars"] = cv.get("review_stars") if cv else None
            v["gnomad_af_popmax"] = gn.get("af_popmax") if gn else None

            acmg = acmg_classify({
                "consequence": v.get("consequence"),
                "impact": v.get("impact"),
                "cadd_score": v.get("cadd_score"),
                "gerp_score": v.get("gerp_score"),
                "gnomad_af_popmax": v.get("gnomad_af_popmax"),
                "clinvar_classification": v.get("clinvar_significance"),
                "clinvar_review_stars": v.get("clinvar_review_stars") or 0,
            })
            v["acmg_tier"] = acmg["tier"]
            v["acmg_evidence_codes"] = [c.value for c in acmg["evidence_codes"]]
            v["acmg_rationale"] = acmg["rationale"]
        return variants
```
